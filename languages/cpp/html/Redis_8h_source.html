221  // helper function for tokenizing redis replies
222  std::vector<std::string> tokenize(std::string s, std::string delimiter) {
223  size_t pos_start = 0, pos_end, delim_len = delimiter.length();
224  std::string token;
225  std::vector<std::string> res;
226 
227  while ((pos_end = s.find(delimiter, pos_start)) != std::string::npos) {
228  token = s.substr(pos_start, pos_end - pos_start);
229  pos_start = pos_end + delim_len;
230  res.push_back(token);
231  }
232 
233  if(!s.substr(pos_start).empty()){
234  res.push_back(s.substr(pos_start));
235  }
236  return res;
237  }

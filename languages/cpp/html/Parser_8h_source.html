149  //flat parser for redis replies - for XADD, XDEL, SET, XACK
150  std::pair<std::shared_ptr<const char *>, size_t>
151  process_flat(const buffer &buff, atom::error & err){
152  logger.debug("process_flat()");
153  std::pair<std::shared_ptr<const char *>, size_t> response;
154  auto end = buf_iter::end(buff.data());
155 
156  for(auto it = buf_iter::begin(buff.data()); it != end; ++it){
157  switch(*it) {
158  case '+': // simple string
159  case '-': // nill or error
160  case ':': { // integer
161  logger.debug("process_flat| simple strings");
162  it++; //move past indicator char
163  size_t str_len = find_data(it, end);
164  logger.debug("process_flat| str_len: " + std::to_string(str_len));
165  std::shared_ptr<const char *> str = std::make_shared<const char *>(&*(it));
166  response = std::pair<std::shared_ptr<const char *>,size_t>(str, str_len);
167  it += (str_len);
168  break;
169  }
170  case '$': {// bulk string - will carry on until /r/n is hit
171  logger.debug("process_flat| bulk strings");
172  it++; //move past the indicator char
173  std::tuple<size_t,size_t> sizes = find_data_len(it, end); // <num_bytes, num characters in data string>
174  size_t num_bytes = std::get<0>(sizes);
175  size_t data_len = std::get<1>(sizes);
176  logger.debug("process_flat| data_len: " + std::to_string(data_len));
177  std::shared_ptr<const char *> data = std::make_shared<const char *>(&*(it+=num_bytes));
178  response = std::pair<std::shared_ptr<const char *>,size_t>(data, data_len);
179  it+= (data_len); //move pointer past the data string
180  break;
181  }
182  }
183  }
184  return response;
185  }
